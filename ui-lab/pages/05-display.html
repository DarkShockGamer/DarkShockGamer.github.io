// Trident Robotics — Auto Scoring Function
function calculateAutoScore(targets, accuracy) {
  const BASE_POINTS = 10;
  const hits = targets.filter(t => t.hit);
  return hits.reduce((total, target) => {
    return total + (BASE_POINTS * target.multiplier * accuracy);
  }, 0);
}
